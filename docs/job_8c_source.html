81/**
82 * @brief Thread entry point: build command, fork/exec re-boot, stream output.
83 *
84 * Owns the heap-allocated @p arg and frees it before doing any blocking
85 * work. Updates the database on start and finish, and signals SSE
86 * connections when the job is complete.
87 *
88 * @param arg Heap-allocated ThreadArg; freed inside this function.
89 * @return Always NULL.
90 */
91static void *runner_thread(void *arg)
92{
93 ThreadArg *ta = arg;
94 int64_t job_id = ta->job_id;
95 char binary[256], upload_dir[256], log_dir[256];
96
97 snprintf(binary, sizeof(binary), "%s", ta->binary);
98 snprintf(upload_dir, sizeof(upload_dir), "%s", ta->upload_dir);
99 snprintf(log_dir, sizeof(log_dir), "%s", ta->log_dir);
100 free(ta);
101
102 /* Fetch job configuration from the database */
103 Job j;
104 if (db_job_get(job_id, &j) != 0)
105 {
106 return NULL;
107 }
108
109 /* Build argv for re-boot */
110 char hex_path[512];
111 snprintf(hex_path, sizeof(hex_path), "%s/%s", upload_dir, j.hex_file);
112
113 char *argv[32];
114 int argc = 0;
115 argv[argc++] = binary;
116 argv[argc++] = "-f"; argv[argc++] = hex_path;
117 argv[argc++] = "-n"; argv[argc++] = j.node_id;
118 argv[argc++] = "-c"; argv[argc++] = j.interface;
119 argv[argc++] = "-i"; argv[argc++] = j.device;
120
121 if (j.tcp_port[0]) { argv[argc++] = "-p"; argv[argc++] = j.tcp_port; }
122 if (j.retries[0]) { argv[argc++] = "-t"; argv[argc++] = j.retries; }
123 if (j.reset_flag[0]) { argv[argc++] = "-r"; argv[argc++] = j.reset_flag; }
124 if (j.verbose[0]) { argv[argc++] = "-v"; argv[argc++] = j.verbose; }
125
126 /* Append extra_args split on spaces (no quoting support) */
127 char extra_buf[DB_STR512];
128 if (j.extra_args[0])
129 {
130 snprintf(extra_buf, sizeof(extra_buf), "%s", j.extra_args);
131 char *tok = strtok(extra_buf, " ");
132 while (tok && argc < 30)
133 {
134 argv[argc++] = tok;
135 tok = strtok(NULL, " ");
136 }
137 }
138 argv[argc] = NULL;
139
140 /* Open log file; log the command line first */
141 char log_path[512];
142 snprintf(log_path, sizeof(log_path), "%s/job_%lld.log",
143 log_dir, (long long)job_id);
144 int log_fd = open(log_path, O_WRONLY | O_CREAT | O_TRUNC, 0644);
145
146 if (log_fd >= 0)
147 {
148 dprintf(log_fd, "CMD:");
149 for (int i = 0; i < argc; i++)
150 {
151 dprintf(log_fd, " %s", argv[i]);
152 }
153 dprintf(log_fd, "\n\n");
154 }
155
156 /* Fork the subprocess, piping its stdout+stderr into the log */
157 int pipefd[2];
158 if (pipe(pipefd) != 0)
159 {
160 db_job_set_finished(job_id, "failed", -1);
161 return NULL;
162 }
163
164 pid_t pid = fork();
165 if (pid == 0)
166 {
167 /* Child: redirect stdout/stderr into the write end of the pipe */
168 close(pipefd[0]);
169 dup2(pipefd[1], STDOUT_FILENO);
170 dup2(pipefd[1], STDERR_FILENO);
171 close(pipefd[1]);
172 execvp(binary, argv);
173 _exit(127);
174 }
175
176 /* Parent: register the running slot, then relay output to the log */
177 close(pipefd[1]);
178
179 pthread_mutex_lock(&g_lock);
180 RunningJob *slot = alloc_job_slot();
181 if (slot)
182 {
183 slot->job_id = job_id;
184 slot->pid = pid;
185 slot->active = 1;
186 }
187 pthread_mutex_unlock(&g_lock);
188
189 db_job_set_started(job_id);
190
191 char buf[1024];
192 ssize_t n;
193 while ((n = read(pipefd[0], buf, sizeof(buf))) > 0)
194 {
195 if (log_fd >= 0)
196 {
197 if (write(log_fd, buf, (size_t)n) != n) { /* best-effort */ }
198 }
199 }
200 close(pipefd[0]);
201 if (log_fd >= 0)
202 {
203 close(log_fd);
204 }
205
206 /* Reap child and record final status */
207 int wstatus = 0;
208 waitpid(pid, &wstatus, 0);
209 int exit_code = WIFEXITED(wstatus) ? WEXITSTATUS(wstatus) : -1;
210 const char *status = (exit_code == 0) ? "success" : "failed";
211
212 pthread_mutex_lock(&g_lock);
213 RunningJob *rj = find_job_slot(job_id);
214 if (rj)
215 {
216 rj->active = 0;
217 }
218 pthread_mutex_unlock(&g_lock);
219
220 db_job_set_finished(job_id, status, exit_code);
221
222 /* Signal SSE connections that the stream has ended */
223 pthread_mutex_lock(&g_lock);
224 for (int i = 0; i < MAX_SSE_CONNS; i++)
225 {
226 if (g_sse[i].active && g_sse[i].job_id == job_id)
227 {
228 g_sse[i].active = 2; /* 2 = flush remaining log then close */
229 }
230 }
231 pthread_mutex_unlock(&g_lock);
232
233 return NULL;
234}

240int job_run(int64_t job_id, const char *binary,
241 const char *upload_dir, const char *log_dir)
242{
243 pthread_mutex_lock(&g_lock);
244 if (find_job_slot(job_id))
245 {
246 pthread_mutex_unlock(&g_lock);
247 return -1;
248 }
249 pthread_mutex_unlock(&g_lock);
250
251 ThreadArg *ta = malloc(sizeof(ThreadArg));
252 ta->job_id = job_id;
253 strncpy(ta->binary, binary, sizeof(ta->binary) - 1);
254 strncpy(ta->upload_dir, upload_dir, sizeof(ta->upload_dir) - 1);
255 strncpy(ta->log_dir, log_dir, sizeof(ta->log_dir) - 1);
256
257 pthread_t tid;
258 pthread_attr_t attr;
259 pthread_attr_init(&attr);
260 pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);
261
262 if (pthread_create(&tid, &attr, runner_thread, ta) != 0)
263 {
264 free(ta);
265 pthread_attr_destroy(&attr);
266 return -1;
267 }
268
269 pthread_attr_destroy(&attr);
270 return 0;
271}

276 RunningJob *rj = find_job_slot(job_id);

294void job_sse_add(int64_t job_id, struct mg_connection *conn)
295{
296 pthread_mutex_lock(&g_lock);
297 for (int i = 0; i < MAX_SSE_CONNS; i++)
298 {
299 if (!g_sse[i].active)
300 {
301 g_sse[i].job_id = job_id;
302 g_sse[i].conn = conn;
303 g_sse[i].log_offset = 0;
304 g_sse[i].active = 1;
305 break;
306 }
307 }
308 pthread_mutex_unlock(&g_lock);
309}

311void job_sse_poll(void)
312{
313 /* g_log_dir is defined in routes.c and declared extern here */
314 extern char g_log_dir[256];
315
316 pthread_mutex_lock(&g_lock);
317 for (int i = 0; i < MAX_SSE_CONNS; i++)
318 {
319 SseConn *sc = &g_sse[i];
320 if (!sc->active)
321 {
322 continue;
323 }
324
325 /* Tail the log file from the offset of the last byte sent */
326 char log_path[512];
327 snprintf(log_path, sizeof(log_path), "%s/job_%lld.log",
328 g_log_dir, (long long)sc->job_id);
329
330 FILE *f = fopen(log_path, "r");
331 if (f)
332 {
333 fseek(f, sc->log_offset, SEEK_SET);
334 char line[1024];
335 while (fgets(line, sizeof(line), f))
336 {
337 size_t len = strlen(line);
338 if (len > 0 && line[len - 1] == '\n')
339 {
340 line[--len] = '\0';
341 }
342 mg_printf(sc->conn, "data: %s\n\n", line);
343 }
344 sc->log_offset = ftell(f);
345 fclose(f);
346 }
347
348 if (sc->active == 2)
349 {
350 /* Job finished — flush any remaining output then close */
351 mg_printf(sc->conn, "data: [STREAM END]\n\n");
352 sc->conn->is_draining = 1;
353 sc->active = 0;
354 }
355 }
356 pthread_mutex_unlock(&g_lock);
357}
