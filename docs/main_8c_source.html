106/**
107 * @brief Program entry point.
108 * @param argc Argument count.
109 * @param argv Argument vector.
110 * @return 0 on clean shutdown, 1 on fatal error.
111 */
112int main(int argc, char *argv[])
113{
114 char listen[64] = "0.0.0.0:5000";
115 char db_path[256] = "reboot.db";
116
117 for (int i = 1; i < argc; i++)
118 {
119 if (!strcmp(argv[i], "-l") && i + 1 < argc)
120 {
121 strncpy(listen, argv[++i], sizeof(listen) - 1);
122 }
123 else if (!strcmp(argv[i], "-b") && i + 1 < argc)
124 {
125 strncpy(g_binary, argv[++i], sizeof(g_binary) - 1);
126 }
127 else if (!strcmp(argv[i], "-d") && i + 1 < argc)
128 {
129 strncpy(db_path, argv[++i], sizeof(db_path) - 1);
130 }
131 else if (!strcmp(argv[i], "-u") && i + 1 < argc)
132 {
133 strncpy(g_upload_dir, argv[++i], sizeof(g_upload_dir) - 1);
134 }
135 else if (!strcmp(argv[i], "-L") && i + 1 < argc)
136 {
137 strncpy(g_log_dir, argv[++i], sizeof(g_log_dir) - 1);
138 }
139 else if (!strcmp(argv[i], "-w") && i + 1 < argc)
140 {
141 strncpy(g_www_dir, argv[++i], sizeof(g_www_dir) - 1);
142 }
143 else if (!strcmp(argv[i], "-h"))
144 {
145 usage(argv[0]);
146 return 0;
147 }
148 else
149 {
150 fprintf(stderr, "Unknown option: %s\n", argv[i]);
151 usage(argv[0]);
152 return 1;
153 }
154 }
155
156 mkdirp(g_upload_dir);
157 mkdirp(g_log_dir);
158
159 if (db_open(db_path) != 0)
160 {
161 fprintf(stderr, "Failed to open database: %s\n", db_path);
162 return 1;
163 }
164 seed_admin("admin", "admin");
165
166 signal(SIGINT, sig_handler);
167 signal(SIGTERM, sig_handler);
168 signal(SIGPIPE, SIG_IGN);
169
170 struct mg_mgr mgr;
171 mg_mgr_init(&mgr);
172
173 if (!mg_http_listen(&mgr, listen, http_handler, NULL))
174 {
175 fprintf(stderr, "Failed to listen on %s\n", listen);
176 db_close();
177 return 1;
178 }
179
180 /* Fire job_sse_poll() every 300 ms to push log lines to SSE clients */
181 mg_timer_add(&mgr, 300, MG_TIMER_REPEAT, sse_timer_cb, NULL);
182
183 printf("\n");
184 printf(" ╔══════════════════════════════════════╗\n");
185 printf(" ║ Re-BOOT Web Server ║\n");
186 printf(" ║ ║\n");
187 printf(" ║ Listening : %-22s ║\n", listen);
188 printf(" ║ Binary : %-22s ║\n", g_binary);
189 printf(" ║ Database : %-22s ║\n", db_path);
190 printf(" ║ Default : admin / admin ║\n");
191 printf(" ╚══════════════════════════════════════╝\n\n");
192
193 while (g_running)
194 {
195 mg_mgr_poll(&mgr, 50);
196 }
197
198 printf("\n[server] Shutting down...\n");
199 mg_mgr_free(&mgr);
200 db_close();
201 return 0;
202}
