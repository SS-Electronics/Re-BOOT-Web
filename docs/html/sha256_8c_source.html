43/**
44 * @brief Process one 512-bit block and update the running hash state.
45 * @param ctx Context carrying the current hash state.
46 * @param data Pointer to exactly 64 bytes of input.
47 */
48static void sha256_transform(SHA256_CTX *ctx, const uint8_t data[])
49{
50 uint32_t a, b, c, d, e, f, g, h, i, j, t1, t2, m[64];
51
52 for (i = 0, j = 0; i < 16; ++i, j += 4)
53 {
54 m[i] = ((uint32_t)data[j] << 24)
55 | ((uint32_t)data[j + 1] << 16)
56 | ((uint32_t)data[j + 2] << 8)
57 | (uint32_t)data[j + 3];
58 }
59 for (; i < 64; ++i)
60 {
61 m[i] = SIG1(m[i - 2]) + m[i - 7] + SIG0(m[i - 15]) + m[i - 16];
62 }
63
64 a = ctx->state[0]; b = ctx->state[1];
65 c = ctx->state[2]; d = ctx->state[3];
66 e = ctx->state[4]; f = ctx->state[5];
67 g = ctx->state[6]; h = ctx->state[7];
68
69 for (i = 0; i < 64; ++i)
70 {
71 t1 = h + EP1(e) + CH(e, f, g) + K[i] + m[i];
72 t2 = EP0(a) + MAJ(a, b, c);
73 h = g; g = f; f = e; e = d + t1;
74 d = c; c = b; b = a; a = t1 + t2;
75 }
76
77 ctx->state[0] += a; ctx->state[1] += b;
78 ctx->state[2] += c; ctx->state[3] += d;
79 ctx->state[4] += e; ctx->state[5] += f;
80 ctx->state[6] += g; ctx->state[7] += h;
81}

112void sha256_final(SHA256_CTX *ctx, uint8_t hash[32])
113{
114 uint32_t i = ctx->datalen;
115
116 if (ctx->datalen < 56)
117 {
118 ctx->data[i++] = 0x80;
119 while (i < 56)
120 {
121 ctx->data[i++] = 0x00;
122 }
123 }
124 else
125 {
126 ctx->data[i++] = 0x80;
127 while (i < 64)
128 {
129 ctx->data[i++] = 0x00;
130 }
131 sha256_transform(ctx, ctx->data);
132 memset(ctx->data, 0, 56);
133 }
134
135 ctx->bitlen += (uint64_t)ctx->datalen * 8;
136 ctx->data[63] = (uint8_t)(ctx->bitlen);
137 ctx->data[62] = (uint8_t)(ctx->bitlen >> 8);
138 ctx->data[61] = (uint8_t)(ctx->bitlen >> 16);
139 ctx->data[60] = (uint8_t)(ctx->bitlen >> 24);
