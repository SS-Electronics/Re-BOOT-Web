17/**
18 * @brief Read @p bytes random bytes from /dev/urandom and write them as
19 * a lowercase hex string (2 * @p bytes chars + NUL) into @p out.
20 * @param bytes Number of random bytes to read (max 64).
21 * @param out Output buffer; must hold at least 2*bytes+1 chars.
22 */
23static void random_hex(int bytes, char *out)
24{
25 uint8_t buf[64];
26 if (bytes > 64)
27 {
28 bytes = 64;
29 }
30
31 int fd = open("/dev/urandom", O_RDONLY);
32 if (fd >= 0)
33 {
34 if (read(fd, buf, (size_t)bytes) < bytes) { /* best-effort */ }
35 close(fd);
36 }
37
38 static const char *h = "0123456789abcdef";
39 for (int i = 0; i < bytes; i++)
40 {
41 out[i * 2] = h[buf[i] >> 4];
42 out[i * 2 + 1] = h[buf[i] & 0xf];
43 }
44 out[bytes * 2] = '\0';
45}

48{
49 char salt[17];

61int auth_check_password(const char *password, const char *stored_hash)
62{
63 char salt[17] = {0};
64 const char *colon = strchr(stored_hash, ':');
65 if (!colon)
66 {
67 return 0;
68 }
69
70 size_t salt_len = (size_t)(colon - stored_hash);
71 if (salt_len >= sizeof(salt))
72 {
73 return 0;
74 }
75 memcpy(salt, stored_hash, salt_len);
76 salt[salt_len] = '\0';
77
78 char buf[512];
79 snprintf(buf, sizeof(buf), "%s:%s", salt, password);
80
81 char hexhash[65];
82 sha256_hex((uint8_t *)buf, strlen(buf), hexhash);
83
84 char expected[DB_STR128];
85 snprintf(expected, sizeof(expected), "%s:%s", salt, hexhash);
86
87 return (strcmp(expected, stored_hash) == 0) ? 1 : 0;
88}

95/**
96 * @brief Parse a "key=value; key2=value2" Cookie header and extract one value.
97 * @param header Pointer to the raw cookie header string.
98 * @param hlen Length of @p header in bytes.
99 * @param name Cookie name to search for (NUL-terminated).
100 * @param val Output buffer for the cookie value.
101 * @param vlen Size of @p val including space for NUL terminator.
102 * @return 1 if the cookie was found, 0 otherwise.
103 */
104static int get_cookie(const char *header, size_t hlen,
105 const char *name, char *val, size_t vlen)
106{
107 size_t nlen = strlen(name);
108 const char *p = header;
109 const char *end = header + hlen;
110
111 while (p < end)
112 {
113 while (p < end && (*p == ' ' || *p == '\t'))
114 {
115 p++;
116 }
117
118 if ((size_t)(end - p) > nlen
119 && memcmp(p, name, nlen) == 0
120 && p[nlen] == '=')
121 {
122 p += nlen + 1;
123 const char *v = p;
124 while (p < end && *p != ';')
125 {
126 p++;
127 }
128 size_t len = (size_t)(p - v);
129 if (len >= vlen)
130 {
131 len = vlen - 1;
132 }
133 memcpy(val, v, len);
134 val[len] = '\0';
135 return 1;
136 }
137
138 while (p < end && *p != ';')
139 {
140 p++;
141 }
142 if (p < end)
143 {
144 p++; /* skip ';' */
145 }
146 }
147 return 0;
148}
