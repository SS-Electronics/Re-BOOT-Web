2 * @file auth.h
3 * @brief Password hashing and session-cookie authentication.
4 *
5 * Passwords are stored as "salt:sha256hex" where the salt is 8 random bytes
6 * encoded as 16 hex characters. Sessions are random 64-char hex tokens
7 * persisted in the SQLite session table with a configurable TTL.
8 *
9 * @author Subhajit Roy <subhajitroy005@gmail.com>
10 * @date 2026-05-06
11 */

16#define SESSION_TTL 86400 /**< Session lifetime in seconds (24 hours) */

18#define COOKIE_NAME "rbsid" /**< HTTP cookie name used to carry session tokens */

20/**
21 * @brief Hash a plaintext password with a fresh random salt.
22 *
23 * Reads TOKEN_LEN/4 bytes from /dev/urandom to generate the salt, then
24 * computes SHA-256("salt:password") and stores the result as
25 * "16-char-salt:64-char-sha256hex" in @p out.
26 *
27 * @param password Plaintext password string (NUL-terminated).
28 * @param out Output buffer; must be at least DB_STR128 bytes.
29 */
30void auth_hash_password(const char *password, char out[DB_STR128]);

32/**
33 * @brief Verify a plaintext password against a stored hash.

35 * @param stored_hash Hash string in "salt:sha256hex" format.
36 * @return 1 if the password matches, 0 otherwise.
37 */
38int auth_check_password(const char *password, const char *stored_hash);
