34/**
35 * @brief Escape a string for safe embedding as a JSON string value.
36 *
37 * Handles backslash, double-quote, and common control characters.
38 *
39 * @param src Source string (NUL-terminated).
40 * @param dst Destination buffer.
41 * @param dlen Size of @p dst in bytes.
42 */
43static void json_esc(const char *src, char *dst, size_t dlen)
44{
45 size_t di = 0;
46 for (size_t si = 0; src[si] && di + 4 < dlen; si++)
47 {
48 unsigned char c = (unsigned char)src[si];
49 if (c == '"')
50 {
51 dst[di++] = '\\'; dst[di++] = '"';
52 }
53 else if (c == '\\')
54 {
55 dst[di++] = '\\'; dst[di++] = '\\';
56 }
57 else if (c == '\n')
58 {
59 dst[di++] = '\\'; dst[di++] = 'n';
60 }
61 else if (c == '\r')
62 {
63 dst[di++] = '\\'; dst[di++] = 'r';
64 }
65 else if (c == '\t')
66 {
67 dst[di++] = '\\'; dst[di++] = 't';
68 }
69 else if (c >= 0x20)
70 {
71 dst[di++] = (char)c;
72 }
73 /* Control characters below 0x20 are silently dropped */
74 }
75 dst[di] = '\0';
76}

79#define JSTR(f) ({ static char _b[512]; json_esc(f, _b, sizeof(_b)); _b; })

135/**
136 * @brief POST /api/login — authenticate and set session cookie.
137 *
138 * Accepts both application/x-www-form-urlencoded and JSON bodies.
139 */
140static void handle_login(struct mg_connection *c, struct mg_http_message *hm)
141{
142 char username[64] = {0}, password[128] = {0};
143
144 mg_http_get_var(&hm->body, "username", username, sizeof(username));
145 mg_http_get_var(&hm->body, "password", password, sizeof(password));
146
147 /* Fall back to JSON body */
148 if (!username[0])
149 {
150 char *tmp;
151 if ((tmp = mg_json_get_str(hm->body, "$.username")) != NULL)
152 {
153 strncpy(username, tmp, sizeof(username) - 1);
154 free(tmp);
155 }
156 if ((tmp = mg_json_get_str(hm->body, "$.password")) != NULL)
157 {
158 strncpy(password, tmp, sizeof(password) - 1);
159 free(tmp);
160 }
161 }
162
163 User u;
164 if (db_user_by_username(username, &u) != 0
165 || !auth_check_password(password, u.pwhash))
166 {
167 reply_err(c, 401, "Invalid credentials");
168 return;
169 }
170
171 char token[65];
172 auth_gen_token(token);
173 db_session_create(token, u.id, (int64_t)time(NULL) + SESSION_TTL);
174
175 char hdrs[256];
176 snprintf(hdrs, sizeof(hdrs),
177 "Content-Type: application/json\r\n"
178 "Set-Cookie: " COOKIE_NAME "=%s; Path=/; HttpOnly; SameSite=Strict\r\n",
179 token);
180
181 char body[1280];
182 snprintf(body, sizeof(body),
183 "{\"ok\":true,\"role\":\"%s\",\"username\":\"%s\"}",
184 JSTR(u.role), JSTR(u.username));
185
186 mg_http_reply(c, 200, hdrs, "%s", body);
187}

293/**
294 * @brief POST /api/jobs — create a job from a multipart form upload.
295 *
296 * The hex_file field must be a file upload. All other fields are optional
297 * text parts. The uploaded file is saved with a Unix-timestamp prefix to
298 * avoid name collisions between concurrent uploads.
299 */
300static void handle_jobs_create(struct mg_connection *c,
301 struct mg_http_message *hm)
302{
303 User *u = require_auth(c, hm);
304 if (!u)
305 {
306 return;
307 }
308
309 Job j;
310 memset(&j, 0, sizeof(j));
311 j.user_id = u->id;
312 free(u);
313
314 struct mg_http_part part;
315 size_t ofs = 0;
316 char hex_filename[256] = {0};
317
318 while ((ofs = mg_http_next_multipart(hm->body, ofs, &part)) > 0)
319 {
320 char fname[64];
321 snprintf(fname, sizeof(fname), "%.*s",
322 (int)part.name.len, part.name.buf);
323
324/* Copy a text field from the multipart part into the corresponding Job field */
325#define GETFIELD(field) \
326 if (strcmp(fname, #field) == 0) \
327 { \
328 size_t l = part.body.len < sizeof(j.field) - 1 \
329 ? part.body.len : sizeof(j.field) - 1; \
330 memcpy(j.field, part.body.buf, l); \
331 j.field[l] = '\0'; \
332 }
333
334 GETFIELD(name)
335 else GETFIELD(node_id)
336 else GETFIELD(interface)
337 else GETFIELD(device)
338 else GETFIELD(tcp_port)
339 else GETFIELD(retries)
340 else GETFIELD(reset_flag)
341 else GETFIELD(verbose)
342 else GETFIELD(extra_args)
343 else if (strcmp(fname, "hex_file") == 0 && part.filename.len > 0)
344 {
345 /* Sanitise the uploaded filename: strip path separators */
346 const char *fn = part.filename.buf;
347 size_t fnlen = part.filename.len;
348 for (size_t k = 0; k < fnlen; k++)
349 {
350 if (fn[k] == '/' || fn[k] == '\\')
351 {
352 fn = fn + k + 1;
353 fnlen = fnlen - k - 1;
354 k = 0;
355 }
356 }
357 size_t l = fnlen < sizeof(hex_filename) - 1
358 ? fnlen : sizeof(hex_filename) - 1;
359 memcpy(hex_filename, fn, l);
360 hex_filename[l] = '\0';
361
362 /* Replace spaces with underscores */
363 for (size_t k = 0; k < strlen(hex_filename); k++)
364 {
365 if (hex_filename[k] == ' ')
366 {
367 hex_filename[k] = '_';
368 }
369 }
370
371 /* Prepend a Unix timestamp to avoid upload collisions */
372 char prefixed[256];
373 snprintf(prefixed, sizeof(prefixed), "%lld_%s",
374 (long long)time(NULL), hex_filename);
375 snprintf(hex_filename, sizeof(hex_filename), "%s", prefixed);
376
377 /* Write the uploaded bytes to disk */
378 char path[512];
379 snprintf(path, sizeof(path), "%s/%s", g_upload_dir, hex_filename);
380 int fd = open(path, O_WRONLY | O_CREAT | O_TRUNC, 0644);
381 if (fd >= 0)
382 {
383 if (write(fd, part.body.buf, part.body.len)
384 != (ssize_t)part.body.len)
385 {
386 /* best-effort; error visible in job log */
387 }
388 close(fd);
389 }
390 snprintf(j.hex_file, sizeof(j.hex_file), "%s", hex_filename);
391 }
392#undef GETFIELD
393 }
394
395 if (!j.hex_file[0] || !j.node_id[0] || !j.device[0])
396 {
397 reply_err(c, 400, "hex_file, node_id and device are required");
398 return;
399 }
400 if (!j.name[0])
401 {
402 strncpy(j.name, "Unnamed Job", sizeof(j.name) - 1);
403 }
404
405 int64_t id = db_job_create(&j);
406 if (id < 0)
407 {
408 reply_err(c, 500, "DB error");
409 return;
410 }
411
412 char body[64];
413 snprintf(body, sizeof(body), "{\"ok\":true,\"id\":%lld}", (long long)id);
414 reply_json(c, 201, body);
415}

701/**
702 * @brief Return 1 if @p uri starts with @p prefix.
703 */
704static int match_prefix(struct mg_str uri, const char *prefix)
705{
706 size_t plen = strlen(prefix);
707 return uri.len >= plen && memcmp(uri.buf, prefix, plen) == 0;

710/**
711 * @brief Extract the integer ID that immediately follows @p prefix in @p uri.
712 * @return Parsed ID, or -1 if the URI is too short.
713 */
714static int64_t parse_id(struct mg_str uri, const char *prefix)
715{
716 size_t plen = strlen(prefix);
717 if (uri.len <= plen)
718 {
719 return -1;
720 }
721 return (int64_t)atoll(uri.buf + plen);
722}

724/**
725 * @brief Return 1 if the path segment after the numeric ID matches @p suffix.
726 * @param uri Full request URI.
727 * @param prefix URI prefix before the numeric ID (e.g. "/api/jobs/").
728 * @param suffix Expected suffix after the ID (e.g. "/run").
729 */
730static int has_suffix(struct mg_str uri, const char *prefix, const char *suffix)
731{
732 size_t plen = strlen(prefix);
733 size_t slen = strlen(suffix);
734
735 if (uri.len < plen)
736 {
737 return 0;
738 }
739
740 /* Advance past the numeric ID segment */
741 size_t i = plen;
742 while (i < uri.len && uri.buf[i] >= '0' && uri.buf[i] <= '9')
743 {
744 i++;
745 }
746
747 if (uri.len - i < slen)
748 {
749 return 0;
750 }
751 return memcmp(uri.buf + i, suffix, slen) == 0;
752}

754/** Test HTTP method against a string literal. */
755#define METH(m) (mg_strcmp(hm->method, mg_str(m)) == 0)
756/** Test URI against a glob pattern. */
757#define URI(p) (mg_match(hm->uri, mg_str(p), NULL))
758/** Test both method and URI. */
759#define IS(m, p) (METH(m) && URI(p))

761void http_handler(struct mg_connection *c, int ev, void *ev_data)
762{
763 if (ev != MG_EV_HTTP_MSG)
764 {
765 return;
766 }
767
768 struct mg_http_message *hm = ev_data;
769 struct mg_str uri = hm->uri;
