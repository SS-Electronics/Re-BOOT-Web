3 * @brief SQLite3 persistence layer — users, jobs, and sessions.

7 * (db_close()). WAL journal mode is enabled for concurrent read access.

16/** @defgroup db_str_sizes Fixed-width string buffer sizes used in DB structs */
17/** @{ */
18#define DB_STR64 64 /**< 64-byte string field */
19#define DB_STR128 128 /**< 128-byte string field */
20#define DB_STR256 256 /**< 256-byte string field */
21#define DB_STR512 512 /**< 512-byte string field */
22/** @} */

24/**
25 * @brief Represents a user account stored in the database.
26 */
27typedef struct
28{
29 int64_t id; /**< Auto-incremented primary key */
30 char username[DB_STR64]; /**< Unique login name */
31 char pwhash[DB_STR128]; /**< Stored as "salt:sha256hex" */
32 char role[16]; /**< "admin" or "user" */
33} User;

35/**
36 * @brief Represents a firmware flash job stored in the database.
37 */
38typedef struct
39{
40 int64_t id; /**< Auto-incremented primary key */
41 char name[DB_STR128]; /**< Human-readable display name */
42 char hex_file[DB_STR256]; /**< Uploaded filename under uploads/ */
43 char node_id[32]; /**< re-boot -n argument */
44 char interface[16]; /**< "serial", "tcp", or "can" */
45 char device[DB_STR128]; /**< re-boot -i argument (path or IP) */
46 char tcp_port[8]; /**< re-boot -p argument */
47 char retries[4]; /**< re-boot -t argument */
48 char reset_flag[4]; /**< re-boot -r argument */
49 char verbose[4]; /**< re-boot -v argument */
50 char extra_args[DB_STR512]; /**< Appended verbatim to command line */
51 char status[16]; /**< "pending","running","success","failed" */
52 char created_at[32]; /**< ISO-8601 creation timestamp */
53 char started_at[32]; /**< ISO-8601 start timestamp */
54 char finished_at[32]; /**< ISO-8601 finish timestamp */
55 int exit_code; /**< Subprocess exit code; -1 if not run */
56 int64_t user_id; /**< Foreign key into user.id */
57 char username[DB_STR64]; /**< Denormalised from joined user row */
58} Job;
